  <script>
    // 初始化：把 data-word 填到 .word 顯示
    document.querySelectorAll('.term-card').forEach(card => {
      const word = card.dataset.word;
      const wordSpan = card.querySelector('.word');
      if (wordSpan) wordSpan.textContent = word;
    });

    function speakTextFromCard(button) {
      const card = button.closest('.term-card');
      const word = card.dataset.word; // 單一來源
      const utterance = new SpeechSynthesisUtterance(word);
      utterance.lang = 'en-US';
      speechSynthesis.speak(utterance);
    }
  </script>
